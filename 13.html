const svgIcon = document.querySelector('.svg-icon svg'); svgIcon.addEventListener('click', function () { const fileInput = document.createElement('input'); fileInput.type = 'file'; fileInput.accept = 'image/*'; fileInput.click(); fileInput.addEventListener('change', function () { if (fileInput.files.length > 0) { const img = document.getElementById('uploadedImage'); img.src = URL.createObjectURL(fileInput.files[0]); img.onload = function () { console.log('Image loaded successfully'); processImage(img); }; } else { console.log('No file selected'); } }); }); function processImage(img) { const canvas = document.getElementById('canvas'); const ctx = canvas.getContext('2d'); const maxWidth = 1024; const maxHeight = 1024; let width = img.width; let height = img.height; if (width > maxWidth || height > maxHeight) { const scale = Math.min(maxWidth / width, maxHeight / height); width *= scale; height *= scale; } canvas.width = width; canvas.height = height; ctx.drawImage(img, 0, 0, width, height); console.log('Image drawn to canvas'); let src = cv.imread(canvas); showDebugImage(src, "Original Uploaded Image"); let gray = new cv.Mat(); cv.cvtColor(src, gray, cv.COLOR_RGBA2GRAY); showDebugImage(gray, "Grayscale Image"); let blurred = new cv.Mat(); cv.GaussianBlur(gray, blurred, new cv.Size(5, 5), 0); showDebugImage(blurred, "Blurred Image"); let threshold = new cv.Mat(); cv.adaptiveThreshold( blurred, threshold, 255, cv.ADAPTIVE_THRESH_GAUSSIAN_C, cv.THRESH_BINARY_INV, 11, 2 ); showDebugImage(threshold, "Thresholded Image"); let contours = new cv.MatVector(); let hierarchy = new cv.Mat(); cv.findContours(threshold, contours, hierarchy, cv.RETR_EXTERNAL, cv.CHAIN_APPROX_SIMPLE); let largestContour = findLargestSquare(contours); if (!largestContour) { console.error("No valid Sudoku grid found."); return; } let approx = new cv.Mat(); cv.approxPolyDP(largestContour, approx, 0.02 * cv.arcLength(largestContour, true), true); if (approx.rows !== 4) { console.error("Sudoku grid was not detected properly."); return; } let sortedPoints = sortCornerPoints(approx); let warped = warpPerspective(src, sortedPoints); showDebugImage(warped, "Warped Sudoku Grid"); extractGridSquares(warped); src.delete(); gray.delete(); blurred.delete(); threshold.delete(); contours.delete(); hierarchy.delete(); largestContour.delete(); approx.delete(); } function findLargestSquare(contours) { let maxArea = 0; let largestContour = null; for (let i = 0; i < contours.size(); i++) { let contour = contours.get(i); let area = cv.contourArea(contour); let peri = cv.arcLength(contour, true); let approx = new cv.Mat(); cv.approxPolyDP(contour, approx, 0.02 * peri, true); if (area > maxArea && approx.rows === 4) { maxArea = area; largestContour = contour.clone(); } approx.delete(); } return largestContour; } function sortCornerPoints(approx) { let points = []; for (let i = 0; i < 4; i++) { points.push({ x: approx.data32S[i * 2], y: approx.data32S[i * 2 + 1] }); } points.sort((a, b) => a.x + a.y - (b.x + b.y)); let topLeft = points[0]; let bottomRight = points[3]; let topRight = points[1].x > points[2].x ? points[1] : points[2]; let bottomLeft = points[1].x < points[2].x ? points[1] : points[2]; return [topLeft, topRight, bottomRight, bottomLeft]; } function warpPerspective(src, points) { let size = 900; let dstMat = cv.matFromArray(4, 1, cv.CV_32FC2, [ 0, 0, size, 0, size, size, 0, size ]); let srcMat = cv.matFromArray(4, 1, cv.CV_32FC2, [ points[0].x, points[0].y, points[1].x, points[1].y, points[2].x, points[2].y, points[3].x, points[3].y ]); let M = cv.getPerspectiveTransform(srcMat, dstMat); let dst = new cv.Mat(); cv.warpPerspective(src, dst, M, new cv.Size(size, size)); srcMat.delete(); dstMat.delete(); M.delete(); return dst; } function findLargestWhiteSquareInCell(cell) { let gray = new cv.Mat(); cv.cvtColor(cell, gray, cv.COLOR_RGBA2GRAY); let threshold = new cv.Mat(); cv.adaptiveThreshold( gray, threshold, 222, cv.ADAPTIVE_THRESH_GAUSSIAN_C, cv.THRESH_BINARY, 15, 5 ); let contours = new cv.MatVector(); let hierarchy = new cv.Mat(); cv.findContours(threshold, contours, hierarchy, cv.RETR_EXTERNAL, cv.CHAIN_APPROX_SIMPLE); let largestContour = null; let maxArea = 0; for (let i = 0; i < contours.size(); i++) { let contour = contours.get(i); let area = cv.contourArea(contour); if (area > maxArea) { let approx = new cv.Mat(); let peri = cv.arcLength(contour, true); cv.approxPolyDP(contour, approx, 0.02 * peri, true); if (approx.rows === 4) { maxArea = area; largestContour = contour.clone(); } approx.delete(); } } if (!largestContour) { return new cv.Rect(0, 0, cell.cols, cell.rows); } let rect = cv.boundingRect(largestContour); let zoomedRect = new cv.Rect( Math.max(0, rect.x + 5), Math.max(0, rect.y + 5), Math.max(0, rect.width - 10), Math.max(0, rect.height - 10) ); contours.delete(); hierarchy.delete(); gray.delete(); threshold.delete(); return zoomedRect; } async function extractGridSquares(grid) { let size = 900; let cellSize = size / 9; let outputSize = 64; let margin = 25; let sudokuNumbers = Array(9).fill().map(() => Array(9).fill([])); for (let row = 0; row < 9; row++) { for (let col = 0; col < 9; col++) { let x = col * cellSize - margin; let y = row * cellSize - margin; let rect = new cv.Rect(x, y, cellSize + 2 * margin, cellSize + 2 * margin); if (x < 0) rect.x = 0; if (y < 0) rect.y = 0; if (x + rect.width > grid.cols) rect.width = grid.cols - x; if (y + rect.height > grid.rows) rect.height = grid.rows - y; let square = grid.roi(rect); let largestSquare = findLargestWhiteSquareInCell(square); let croppedSquare = square.roi(largestSquare); let resized = new cv.Mat(); cv.resize(croppedSquare, resized, new cv.Size(outputSize, outputSize)); let brightened = new cv.Mat(); cv.convertScaleAbs(resized, brightened, 1, 50); let contrasted = new cv.Mat(); cv.convertScaleAbs(brightened, contrasted, 1.5, 0); let blob = await matToBlob(contrasted); let recognizedNumber = await recognizeDigit(blob); sudokuNumbers[row][col] = recognizedNumber === 0 ? [null] : [recognizedNumber]; showDebugImage(contrasted, `Cell (${row},${col}): ${recognizedNumber}`); square.delete(); croppedSquare.delete(); resized.delete(); } } console.log("Extracted Sudoku Grid:", sudokuNumbers); return sudokuNumbers; } function matToBlob(mat) { return new Promise((resolve) => { let canvas = document.createElement('canvas'); canvas.width = mat.cols; canvas.height = mat.rows; let ctx = canvas.getContext('2d'); let imgData = new ImageData(new Uint8ClampedArray(mat.data), mat.cols, mat.rows); ctx.putImageData(imgData, 0, 0); canvas.toBlob((blob) => resolve(blob), 'image/png'); }); } async function recognizeDigit(blob) { const { data: { text } } = await Tesseract.recognize( blob, 'eng', { tessedit_char_whitelist: '123456789', classify_bln_numeric_mode: 1 } ); let digit = text.trim(); return digit.length === 1 && !isNaN(digit) ? parseInt(digit, 10) : 0; } function showDebugImage(mat, title) { let debugCanvas = document.createElement('canvas'); debugCanvas.width = mat.cols; debugCanvas.height = mat.rows; document.body.appendChild(debugCanvas); let ctx = debugCanvas.getContext('2d'); let displayMat = new cv.Mat(); if (mat.type() === cv.CV_8UC1) { cv.cvtColor(mat, displayMat, cv.COLOR_GRAY2RGBA); } else { cv.cvtColor(mat, displayMat, cv.COLOR_BGR2RGBA); } let imgData = new ImageData(new Uint8ClampedArray(displayMat.data), displayMat.cols, displayMat.rows); ctx.putImageData(imgData, 0, 0); let caption = document.createElement('p'); caption.textContent = title; document.body.appendChild(caption); displayMat.delete(); }
